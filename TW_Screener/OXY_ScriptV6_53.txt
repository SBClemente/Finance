f_pad_R(s, w) =>
    string out = str.tostring(s)
    int diff = w - str.length(out)
    diff > 0 ? out + str.substring("                                                  ", 0, diff) : out

f_pad_L(s, w) =>
    string out = str.tostring(s)
    int diff = w - str.length(out)
    diff > 0 ? str.substring("                                                  ", 0, diff) + out : out

fM(v) => na(v) ? f_pad_L("-", 8) : f_pad_L(str.tostring(v / 1000000, '#.0'), 8)

get_base_score(c_ev, c_fy, c_nd, c_fc, c_cg) =>
    tw = (pts(c_ev)>0 ? 0.25 : 0) + (pts(c_fy)>0 ? 0.0 : 0) + (pts(c_nd)>0 ? 0.25 : 0) + (pts(c_fc)>0 ? 0.30 : 0) + (pts(c_cg)>0 ? 0.20 : 0) // SCORE REBALANCEADO
    tp = (pts(c_ev)>0 ? pts(c_ev)*0.25 : 0) + (pts(c_fy)>0 ? pts(c_fy)*0.0 : 0) + (pts(c_nd)>0 ? pts(c_nd)*0.25 : 0) + (pts(c_fc)>0 ? pts(c_fc)*0.30 : 0) + (pts(c_cg)>0 ? pts(c_cg)*0.20 : 0)
    tw > 0 ? tp / (tw * 5.0) : na

calc_oxy_tradeoffs(ebitda_p, ebitda_av, fcf_, fcf_av, eve, fy, fc, cagr, fy_av, fc_av, de, de_av, gm, gm_av, f_vol, eve_av, fc_1, fc_2) =>
    bool rule_sev_div_cx = not na(ebitda_p) and not na(ebitda_av) and not na(fcf_) and not na(fcf_av) and not na(fc) and not na(fc_1) and ebitda_p > ebitda_av and fcf_ < (fcf_av * 0.60) and fc < fc_av * 0.75 and fc_1 < fc_av * 0.75
    bool rule_ult_trap   = not na(eve) and not na(fy) and not na(fc) and not na(fc_1) and not na(cagr) and eve < 8 and fy < 0.05 and fc < 0.35 and fc_1 < 0.40 and cagr < 0
    bool rule_div_cx     = not na(ebitda_p) and not na(ebitda_av) and not na(fcf_) and not na(fcf_av) and not na(fc_1) and ebitda_p > ebitda_av and fcf_ < (fcf_av * 0.85) and fcf_ >= (fcf_av * 0.60) and fc_1 < fc_av * 0.90
    bool rule_val_trap   = not na(eve) and not na(fy) and not na(cagr) and eve < 8 and fy < 0.05 and cagr < 0.04 and not rule_ult_trap
    bool rule_liq_mode   = not na(fc) and not na(fc_av) and not na(cagr) and fc > fc_av and cagr < -0.03
    bool rule_debt_fund  = not na(fc) and not na(fc_av) and not na(de) and not na(de_av) and fc > fc_av and de > de_av and (de - de_av) > 0.5
    bool rule_bought_rev = not na(cagr) and not na(gm) and not na(gm_av) and cagr > 0.05 and gm < (gm_av - 0.05)

calc_yo_loc(curr, prev) => float(na(curr) or na(prev) or prev <= 0 ? na : (curr - prev) / math.abs(prev))
calc_cagr_loc(v_end, v_start) => float(na(v_start) or na(v_end) or v_start <= 0 or v_end <= 0 ? na : math.pow(v_end / v_start, 0.2) - 1)

// FUNÇÃO GLOBAL - Soma de FCF (Progressão Geométrica)
get_acc_fcf(f0, g, yrs) => g == 0 ? f0 * yrs : f0 * (1.0 + g) * (math.pow(1.0 + g, yrs) - 1.0) / g

get_avg5_loc(arr) =>
    s = 0.0, n = 0
    for i = 0 to 4
        v = array.get(arr, i)
        if not na(v)
            s += v, n += 1
    n > 0 ? s / n : na

get_vol_loc(arr) =>
    vals = array.new_float(0)
    for i = 0 to 3
        v = array.get(arr, i)
        if not na(v) and v != 0
            array.push(vals, v)
    res = float(na)
    if array.size(vals) >= 3
        mx = array.max(vals), mn = array.min(vals)
        if mx > 0
            res := (mx - mn) / mx
    res
